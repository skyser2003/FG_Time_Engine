CMAKE_MINIMUM_REQUIRED (VERSION 2.6)
PROJECT (Time_Engine)

SET (TIMEENGINE_SOURCE_DIR "${PROJECT_SOURCE_DIR}/source")
SET (TIMEENGINE_PCH_SUBDIR ${TIMEENGINE_SOURCE_DIR}/StdAfx)
ADD_SUBDIRECTORY(${TIMEENGINE_SOURCE_DIR})

FILE(GLOB TIMEENGINE_SOURCE_SUBDIR RELATIVE ${TIMEENGINE_SOURCE_DIR} ${TIMEENGINE_SOURCE_DIR}/*)
FOREACH(child ${TIMEENGINE_SOURCE_SUBDIR})
	MESSAGE(${child})
	IF(IS_DIRECTORY ${TIMEENGINE_SOURCE_DIR}/${child})
		ADD_SUBDIRECTORY(${TIMEENGINE_SOURCE_DIR}/${child})
	ENDIF()
ENDFOREACH()

INCLUDE_DIRECTORIES( ${TIMEENGINE_SOURCE_DIR} )

ADD_LIBRARY (TimeEngine
	${TIMEENGINE_SOURCE_FILES}
	)
	
SOURCE_GROUP(TimeEngine
	FILES ${TIMEENGINE_SOURCE_FILES}
	)
	
SET_TARGET_PROPERTIES(TimeEngine PROPERTIES LINKER_LANGUAGE C++)
